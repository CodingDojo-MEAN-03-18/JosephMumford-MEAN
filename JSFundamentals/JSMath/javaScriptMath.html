JavaScript Math Functions

                function zero_negativity(numbers){
                    value = false;
                    for(var i = 0; i < numbers.length; i++){
                        if(numbers[i] == 0){
                            value = true;
                        }
                    }
                    return value;
                }
        
                function is_even(number){
                    if(number % 2 == 0){
                        return true;
                    }
                    else{
                        return false;
                    }
                }
        
                function how_many_even(numbers){
                    var value = 0;
                    for(var i = 0; i < numbers.length; i++){
                        if(is_even(numbers[i])){
                            value++;
                        }
                    }
                    return value;
                }
        
                function create_dummy_array(n){
                    var value = [];
                    for(var i = 0; i < n; i++){
                        value.push(get_random_int(9))
                    }
                    return value;
                }
        
                function random_choice(numbers){
                    var value = get_random_int(numbers.length);
                    return numbers[value];
                }
    
                function get_random_int(max){
                    return Math.floor(Math.random() * Math.floor(max));
                }
